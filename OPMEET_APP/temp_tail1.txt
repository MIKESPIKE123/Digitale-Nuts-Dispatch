        const { csX, csY, offsetX, offsetY, cw, ch } = getGridMetrics();
        const components = [
          ...(settings.layerV ? findComponents('V') : []),
          ...(settings.layerH ? findComponents('H') : []),
        ];

        ctx.save();
        ctx.translate(offsetX, offsetY);
        ctx.font = '12px Arial';
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';

        components.forEach(comp => {
            const widthM = (comp.maxC - comp.minC + 1) * cw;
            const heightM = (comp.maxR - comp.minR + 1) * ch;

            const centerX = (comp.minC + (comp.maxC - comp.minC + 1) / 2) * csX;
            const centerY = (comp.minR + (comp.maxR - comp.minR + 1) / 2) * csY;

            const topY = comp.minR * csY;
            const bottomY = (comp.maxR + 1) * csY;
            const leftX = comp.minC * csX;
            const rightX = (comp.maxC + 1) * csX;

            // Draw horizontal dimension line
            ctx.beginPath();
            ctx.moveTo(leftX, topY - 15);
            ctx.lineTo(rightX, topY - 15);
            ctx.moveTo(leftX, topY - 20);
            ctx.lineTo(leftX, topY - 10);
            ctx.moveTo(rightX, topY - 20);
            ctx.lineTo(rightX, topY - 10);
            ctx.strokeStyle = '#000';
            ctx.stroke();
            ctx.fillStyle = '#000';
            ctx.fillText(`${widthM.toFixed(2)} m`, centerX, topY - 25);

            // Draw vertical dimension line
            ctx.beginPath();
            ctx.moveTo(leftX - 15, topY);
            ctx.lineTo(leftX - 15, bottomY);
            ctx.moveTo(leftX - 20, topY);
            ctx.lineTo(leftX - 10, topY);
            ctx.moveTo(leftX - 20, bottomY);
            ctx.lineTo(leftX - 10, bottomY);
            ctx.stroke();
            ctx.save();
            ctx.translate(leftX - 25, centerY);
            ctx.rotate(-Math.PI / 2);
            ctx.fillText(`${heightM.toFixed(2)} m`, 0, 0);
            ctx.restore();
        });

        ctx.restore();
    }

  function drawComponentLabels() {
    const { csX, csY, offsetX, offsetY, cw, ch } = getGridMetrics();
    const comps = [
      ...(settings.layerV ? findComponents('V').map(c => ({ ...c, t: 'V' })) : []),
      ...(settings.layerH ? findComponents('H').map(c => ({ ...c, t: 'H' })) : [])
    ];
    ctx.save();
    ctx.translate(offsetX, offsetY);
    ctx.font = '12px Arial';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    comps.forEach(comp => {
      const widthM = (comp.maxC - comp.minC + 1) * cw;
      const heightM = (comp.maxR - comp.minR + 1) * ch;
      const area = widthM * heightM;
      const centerX = (comp.minC + (comp.maxC - comp.minC + 1) / 2) * csX;
      const centerY = (comp.minR + (comp.maxR - comp.minR + 1) / 2) * csY;
      const text = `${comp.t} • ${widthM.toFixed(2)}×${heightM.toFixed(2)} m • ${area.toFixed(2)} m²`;
      const pad = 6;
      const metrics = ctx.measureText(text);
      const boxW = metrics.width + pad * 2;
      const boxH = 20 + pad * 2;
      ctx.fillStyle = 'rgba(255,255,255,0.05)'; // maximaal transparant
      ctx.strokeStyle = 'rgba(0,0,0,0.08)';
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.rect(centerX - boxW / 2, centerY - boxH / 2, boxW, boxH);
      ctx.fill();
      ctx.stroke();
      ctx.fillStyle = '#111827';
      ctx.fillText(text, centerX, centerY);
    });
    ctx.restore();
  }

    // === Buffer Functions ===
  function computeRegMasks(cw, ch, bufferM, mode) {
        const sub = Math.max(2, parseInt(settings.microSub || 8));
        const microRows = rows * sub;
        const microCols = cols * sub;
        const mx = cw / sub;
        const my = ch / sub;
        const microArea = mx * my;

        const maskV = new Uint8Array(microRows * microCols);
        const maskH = new Uint8Array(microRows * microCols);
  const coreV = new Uint8Array(microRows * microCols);
  const coreH = new Uint8Array(microRows * microCols);

    function setRect(mask, r0, c0, r1, c1) {
            const rs = Math.max(0, Math.floor(r0));
            const re = Math.min(microRows - 1, Math.floor(r1));
            const cs = Math.max(0, Math.floor(c0));
            const ce = Math.min(microCols - 1, Math.floor(c1));
            for (let rr = rs; rr <= re; rr++) {
                let idx = rr * microCols + cs;
                for (let cc = cs; cc <= ce; cc++, idx++) {
                    mask[idx] = 1;
                }
            }
        }

    // Harde clipping op rooi-/voetlijnen: split een rechthoek zodat die niet over een lijn gaat
    const rooiType = rooiTypeEl?.value || 'none';
    const voetType = voetTypeEl?.value || 'none';
    const rooiPos = parseBE(rooiPosEl?.value || '0');
    const voetPos = parseBE(voetPosEl?.value || '0');
    const vLines = [];
    const hLines = [];
    if (rooiType === 'vertical') vLines.push(rooiPos);
    if (rooiType === 'horizontal') hLines.push(rooiPos);
    if (voetType === 'vertical') vLines.push(voetPos);
    if (voetType === 'horizontal') hLines.push(voetPos);

    function addRectClippedForComp(comp, mask, r0, c0, r1, c1) {
      const EPS = 1e-9;
      // Start met één rechthoek; splits langs elke lijn, maar behoud enkel de kant
      // waar de component zich bevindt. Bij een component dat de lijn kruist, behoud beide.
      let rects = [{ r0, c0, r1, c1 }];
      const leftX = comp.minC * cw;
      const rightX = (comp.maxC + 1) * cw;
      const topY = comp.minR * ch;
      const bottomY = (comp.maxR + 1) * ch;

      // Verticale splits en filtering
      for (const L of vLines) {
        const compLeft = rightX <= L - EPS;   // component volledig links
        const compRight = leftX >= L + EPS;   // component volledig rechts
        const touchLeft = Math.abs(rightX - L) <= EPS;   // raakt lijn aan de rechterrand
        const touchRight = Math.abs(leftX - L) <= EPS;   // raakt lijn aan de linkerrand
        const cutLeft = Math.floor((L - EPS) / mx);    // laatste kolom strikt links
        const rightStart = Math.ceil(L / mx);          // eerste kolom strikt rechts
        const newRects = [];
        for (const rc of rects) {
          // Links deel
          const c0L = rc.c0;
          const c1L = Math.min(rc.c1, cutLeft);
          const keepLeft = compLeft || (!compLeft && !compRight && (touchLeft || (!touchLeft && !touchRight))); // als tangent links of echt kruis
          if (keepLeft && c0L <= c1L) {
            newRects.push({ r0: rc.r0, c0: c0L, r1: rc.r1, c1: c1L });
          }
          // Rechts deel
          const c0R = Math.max(rc.c0, rightStart);
          const c1R = rc.c1;
          const keepRight = compRight || (!compLeft && !compRight && (touchRight || (!touchLeft && !touchRight)));
          if (keepRight && c0R <= c1R) {
            newRects.push({ r0: rc.r0, c0: c0R, r1: rc.r1, c1: c1R });
          }
        }
        rects = newRects;
      }
      // Horizontale splits en filtering
      for (const Ly of hLines) {
        const compTop = bottomY <= Ly - EPS;   // component volledig boven
        const compBottom = topY >= Ly + EPS;   // component volledig onder
        const touchTop = Math.abs(bottomY - Ly) <= EPS;   // raakt lijn aan onderrand
        const touchBottom = Math.abs(topY - Ly) <= EPS;   // raakt lijn aan bovenrand
        const cutTop = Math.floor((Ly - EPS) / my);     // laatste rij strikt boven
        const bottomStart = Math.ceil(Ly / my);         // eerste rij strikt onder
        const newRects = [];
        for (const rc of rects) {
          // Boven deel
          const r0T = rc.r0;
          const r1T = Math.min(rc.r1, cutTop);
          const keepTop = compTop || (!compTop && !compBottom && (touchTop || (!touchTop && !touchBottom)));
          if (keepTop && r0T <= r1T) {
            newRects.push({ r0: r0T, c0: rc.c0, r1: r1T, c1: rc.c1 });
          }
          // Onder deel
          const r0B = Math.max(rc.r0, bottomStart);
          const r1B = rc.r1;
          const keepBottom = compBottom || (!compTop && !compBottom && (touchBottom || (!touchTop && !touchBottom)));
          if (keepBottom && r0B <= r1B) {
            newRects.push({ r0: r0B, c0: rc.c0, r1: r1B, c1: rc.c1 });
          }
        }
        rects = newRects;
      }
      for (const rc of rects) setRect(mask, rc.r0, rc.c0, rc.r1, rc.c1);
    }

    const compsV = findComponents('V');
    const compsH = findComponents('H');

    // Markeer kern (exacte celinhoud) altijd in coreV/coreH
    function markCore(comp, coreMask) {
      for (const [r, c] of comp.cells) {
        const baseR = r * sub, baseC = c * sub;
        for (let dr = 0; dr < sub; dr++) {
          for (let dc = 0; dc < sub; dc++) {
            const mr = baseR + dr, mc = baseC + dc;
            coreMask[mr * microCols + mc] = 1;
          }
        }
      }
    }
    for (const comp of compsV) markCore(comp, coreV);
    for (const comp of compsH) markCore(comp, coreH);
    // Boordsteen: markeer kern in beide cores (blokkeert beide maskers), geen eigen offset
    for (let r = 0; r < rows; r++) {
      for (let c = 0; c < cols; c++) {
        if (grid[r][c] === 'B') {
          const baseR = r * sub, baseC = c * sub;
          for (let dr = 0; dr < sub; dr++) {
            for (let dc = 0; dc < sub; dc++) {
              const mr = baseR + dr, mc = baseC + dc;
              const idx = mr * microCols + mc;
              coreV[idx] = 1;
              coreH[idx] = 1;
            }
          }
        }
      }
    }

    if (mode === 'strip') {
      // Union van per-cel rechthoeken met offset (L∞-buffer rond formele vorm), geen volledige component-boundingbox
      const padY = Math.ceil(bufferM / my);
      const padX = Math.ceil(bufferM / mx);
      for (const comp of compsV) {
        for (const [r, c] of comp.cells) {
          const r0 = r * sub - padY;
          const c0 = c * sub - padX;
          const r1 = (r + 1) * sub - 1 + padY;
          const c1 = (c + 1) * sub - 1 + padX;
          addRectClippedForComp(comp, maskV, r0, c0, r1, c1);
        }
      }
      for (const comp of compsH) {
        for (const [r, c] of comp.cells) {
          const r0 = r * sub - padY;
          const c0 = c * sub - padX;
          const r1 = (r + 1) * sub - 1 + padY;
          const c1 = (c + 1) * sub - 1 + padX;
          addRectClippedForComp(comp, maskH, r0, c0, r1, c1);
        }
      }
    } else {
            // Exact mode
      function markExact(comp, mask) {
                const prim = [];
                for (const [r, c] of comp.cells) {
                    const baseR = r * sub, baseC = c * sub;
                    for (let dr = 0; dr < sub; dr++) {
                        for (let dc = 0; dc < sub; dc++) {
                            const mr = baseR + dr, mc = baseC + dc;
                            mask[mr * microCols + mc] = 1;
                            prim.push({ x: (mc + 0.5) * mx, y: (mr + 0.5) * my });
                        }
                    }
                }
                const minMR = Math.max(0, comp.minR * sub - Math.ceil(bufferM / my));
                const maxMR = Math.min(microRows - 1, (comp.maxR + 1) * sub - 1 + Math.ceil(bufferM / my));
                const minMC = Math.max(0, comp.minC * sub - Math.ceil(bufferM / mx));
                const maxMC = Math.min(microCols - 1, (comp.maxC + 1) * sub - 1 + Math.ceil(bufferM / mx));
                const buf2 = bufferM * bufferM;
        const EPS = 1e-9;
        const sgn = (a) => (Math.abs(a) <= EPS ? 0 : (a > 0 ? 1 : -1));
        // Kandidaten exact op een lijn niet automatisch uitsluiten; laat side-test beslissen
        function isOnAnyLine(x, y) {
          return false;
        }
        function sameSideAllLines(px, py, x, y) {
          for (const L of vLines) {
            const sp = sgn(px - L);
            const sc = sgn(x - L);
            if (sp === 0) return false; // vermijd ambigue prim op lijn
            if (sc === 0) continue;     // centrum op lijn toestaan, geen crossing
            if (sp !== sc) return false;
          }
          for (const Ly of hLines) {
            const sp = sgn(py - Ly);
            const sc = sgn(y - Ly);
            if (sp === 0) return false;
            if (sc === 0) continue;
            if (sp !== sc) return false;
          }
          return true;
        }
                for (let mr = minMR; mr <= maxMR; mr++) {
                    const y = (mr + 0.5) * my;
                    for (let mc = minMC; mc <= maxMC; mc++) {
                        const idx = mr * microCols + mc;
                        if (mask[idx]) continue;
                        const x = (mc + 0.5) * mx;
            // Niet langer automatisch overslaan bij exact op lijn
                        let minD2 = Infinity;
                        for (let k = 0; k < prim.length; k++) {
              const p = prim[k];
              if (!sameSideAllLines(p.x, p.y, x, y)) continue;
              const dx = x - p.x;
              const dy = y - p.y;
              const d2 = dx * dx + dy * dy;
              if (d2 < minD2) minD2 = d2;
              if (minD2 <= buf2) break;
                        }
                        if (minD2 <= buf2) mask[idx] = 1;
                    }
                }
            }
            for (const comp of compsV) markExact(comp, maskV);
            for (const comp of compsH) markExact(comp, maskH);
        }

    // Extra regel: voeg 0.50 m band toe tot rooi-/voetlijn wanneer component binnen 0.50 m ligt
  const lineBandM = typeof settings.nearLineBand === 'number' ? settings.nearLineBand : 0.50;
    const totalW = cols * cw;
    const totalH = rows * ch;

    function addBandToLine(comp, mask, lineType, linePosM) {
      if (!lineType || lineType === 'none') return;
      // component bbox in meters
      const leftX = comp.minC * cw;
      const rightX = (comp.maxC + 1) * cw;
      const topY = comp.minR * ch;
      const bottomY = (comp.maxR + 1) * ch;
      if (lineType === 'vertical') {
        const Lx = linePosM;
        if (Lx < 0 || Lx > totalW) return;
        let d = 0;
        if (Lx < leftX) d = leftX - Lx; else if (Lx > rightX) d = Lx - rightX; else d = 0;
        if (d > 0 && d < lineBandM) {
          const x0 = Math.max(0, Math.min(Lx, leftX));
          const x1 = Math.min(totalW, Math.max(Lx, leftX));
          const y0 = Math.max(0, topY);
          const y1 = Math.min(totalH, bottomY);
          const c0 = Math.floor(x0 / mx);
          const c1 = Math.min(microCols - 1, Math.floor((x1 - 1e-9) / mx));
          const r0 = Math.floor(y0 / my);
          const r1 = Math.min(microRows - 1, Math.floor((y1 - 1e-9) / my));
          addRectClippedForComp(comp, mask, r0, c0, r1, c1);
        }
      } else if (lineType === 'horizontal') {
        const Ly = linePosM;
        if (Ly < 0 || Ly > totalH) return;
        let d = 0;
        if (Ly < topY) d = topY - Ly; else if (Ly > bottomY) d = Ly - bottomY; else d = 0;
        if (d > 0 && d < lineBandM) {
          const y0 = Math.max(0, Math.min(Ly, topY));
          const y1 = Math.min(totalH, Math.max(Ly, topY));
          const x0 = Math.max(0, leftX);
          const x1 = Math.min(totalW, rightX);
          const r0 = Math.floor(y0 / my);
          const r1 = Math.min(microRows - 1, Math.floor((y1 - 1e-9) / my));
          const c0 = Math.floor(x0 / mx);
          const c1 = Math.min(microCols - 1, Math.floor((x1 - 1e-9) / mx));
          addRectClippedForComp(comp, mask, r0, c0, r1, c1);
        }
      }
    }

    for (const comp of compsV) {
      addBandToLine(comp, maskV, rooiType, rooiPos);
      addBandToLine(comp, maskV, voetType, voetPos);
    }
    for (const comp of compsH) {
      addBandToLine(comp, maskH, rooiType, rooiPos);
      addBandToLine(comp, maskH, voetType, voetPos);
    }

    // Clip: geen V-offset in H-kern, geen H-offset in V-kern
    for (let i = 0; i < microRows * microCols; i++) {
      if (coreH[i]) maskV[i] = 0;
      if (coreV[i]) maskH[i] = 0;
    }
    // Overlapregel: afhankelijk van beleid: 'V_wins' of 'H_wins'
    const policy = settings.overlapPolicy || 'V_wins';
    if (policy === 'V_wins') {
      for (let i = 0; i < microRows * microCols; i++) {
        if (maskV[i] && maskH[i]) maskH[i] = 0;
      }
    } else if (policy === 'H_wins') {
      for (let i = 0; i < microRows * microCols; i++) {
        if (maskV[i] && maskH[i]) maskV[i] = 0;
      }
    } else if (policy === 'clip_both') {
      for (let i = 0; i < microRows * microCols; i++) {
        if (maskV[i] && maskH[i]) { maskV[i] = 0; maskH[i] = 0; }
      }
    }

  return { maskV, maskH, coreV, coreH, microRows, microCols, sub, microArea, compsV, compsH };
    }
